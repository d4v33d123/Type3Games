175 void Credits::calculateFPS()
176 {
177  static const int NUM_SAMPLES = 1000;
178  static float frameTimes[NUM_SAMPLES];
179  static int currentFrame = 0;
180  static float prevTicks = SDL_GetTicks();
181 
182  float currentTicks;
183  currentTicks = SDL_GetTicks();
184 
185  frameTime_ = currentTicks - prevTicks;
186 
187  //avoid bug that set frameTime_ to very low negative number on first calculation
188  if(frameTime_ < 0)
189  frameTime_ = 0;
190 
191  frameTimes[currentFrame % NUM_SAMPLES] = frameTime_;
192 
193  // set previous ticks to new ticks now
194  prevTicks = currentTicks;
195 
196  currentFrame++;
197 
198  int count;
199 
200  if (currentFrame < NUM_SAMPLES)
201  {
202  count = currentFrame;
203  }
204  else
205  {
206  count = NUM_SAMPLES;
207  }
208 
209  float frameTimeAverage = 0;
210  for (int i = 0; i < count; i++)
211  {
212  frameTimeAverage += frameTimes[i];
213  }
214  frameTimeAverage /= count;
215 
216  if (frameTimeAverage > 0)
217  {
218  fps_ = 1000.0 / frameTimeAverage;
219  }
220  else
221  {
222  fps_ = 60.0f;
223  }
