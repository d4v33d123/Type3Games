330 command MainGame::gameLoop()
331 {
332  Uint32 old_ticks = 0;
333  Uint32 ticks = 0;
334 
335  timer_.set( 3000 );
336 
337  //our game loop
338  while( gameState_ != GameState::EXIT )
339  {
340  // used for frame time measuring
341  float startTicks = SDL_GetTicks();
342  calculateFPS();
343 
344  if( grid_.getCurrency() > 0 )
345  {
346  if( !paused_ )
347  {
348  if( grid_.update( frameTime_, world_to_grid(touch_to_world(pressPos_)) ) )
349  cellSelected_ = false;
350 
351  if(grid_.playVessel())
352  {
353  bloodV_.play();
354  grid_.resetPlayVessel();
355  }
356 
357  // Count the number of cancer cells
358  int num_cancer_cells = 0;
359  for( int i = 0; i < grid_.numCells(); ++i ) {
360  if( ((T3E::Cell*)grid_.getCell(i)->getNode())->getState() == T3E::CellState::CANCEROUS ) num_cancer_cells++;
361  }
362 
363  // if a second has passed, reduce the score by the score per cancer per second
364  if( old_ticks / 1000 < ticks / 1000 ) {
365  grid_.addScore( num_cancer_cells * T3E::SCORE::CANCER_PER_SECOND() );
366  }
367  }
368  }
369  else // game over!!!
370  {
371  gameOver_ = true;
372  textRenderer_.putString( "Game over!", -0.5, 0.3, 100 );
373  }
374 
375  score_ = grid_.getHighScore();
376  textRenderer_.putNumber( grid_.getHighScore() * 100, 10, -0.65, 0.94, 44 );
377  textRenderer_.putNumber( grid_.getCurrency(), 10, -0.9, 0.79, 44 );
378 
379  // Render the tutorial text
380  if( tutorial_ )
381  renderTutorial();
382 
383  renderGame();
384 
385  processInput(frameTime_);
386 
387  float frameTicks = SDL_GetTicks() - startTicks;
388  //Limit the FPS to the max FPS
389  if (1000.0f / maxFPS_ > frameTicks)
390  {
391  SDL_Delay(1000.0f / maxFPS_ - frameTicks);
392  }
393 
394  old_ticks = ticks;
395  ticks = SDL_GetTicks();
396  }
397 
398  return command::MENU;
399 }

981 glm::vec4 MainGame::touch_to_world( glm::vec2 touch_coord )
982 {
983  glm::vec4 result( touch_coord.x, touch_coord.y, 0.0f, 1.0f );
984 
985  result.x = result.x * 2.0f - 1.0f;
986  result.y = result.y * 2.0f - 1.0f;
987  result.y *= -1.0f; // Invert to match OpenGL coords
988 
989  // Calculate the inverse matrix of the view and projection
990  viewProjInverse = glm::inverse( projectionM_ * viewM_ );
991 
992  // multiply the touch position by that
993  result = viewProjInverse * result;
994 
995  // divide by w
996  result.x /= result.w;
997  result.y /= result.w;
998 
999  // scale the position to account for zoom
1000  result.x = camera_.getPosition().x + ( result.x - camera_.getPosition().x ) * result.w * camera_.getPosition().z;
1001  result.y = camera_.getPosition().y + ( result.y - camera_.getPosition().y ) * result.w * camera_.getPosition().z;
1002 
1003  return result;
1004 }

1006 SDL_Point MainGame::world_to_grid( glm::vec4 world_coord )
1007 {
1008  float fracCol, fracRow, fracZ;//fractional coordinates
1009  int col, row, z;//final coordinates
1010  float colD, rowD, zD;//difference between fractional and rounded coords
1011 
1012  glm::mat2 layoutInverse_= glm::mat2(sqrt(3.0f) / 3.0f, -1.0f / 3.0f, 0.0f, 2.0f / 3.0f);
1013  fracCol = (layoutInverse_[0][0] * world_coord.x + layoutInverse_[0][1] * world_coord.y) / 0.54f;
1014  fracRow = (layoutInverse_[1][0] * world_coord.x + layoutInverse_[1][1] * world_coord.y) / 0.54f;
1015 
1016  fracZ = -fracCol -fracRow;
1017 
1018  col = round(fracCol);
1019  row = round(fracRow);
1020  z = round(fracZ);
1021 
1022  colD = abs(col - fracCol);
1023  rowD = abs(row - fracRow);
1024  zD = abs(z - fracZ);
1025 
1026  if((colD > rowD) && (colD > zD))
1027  {
1028  col = -row-z;
1029  }
1030  else if(rowD > zD)
1031  {
1032  row = -col-z;
1033  }
1034  else
1035  {
1036  z = -row-col;
1037  }
1038 
1039  return SDL_Point{ row, col};
1040 }

1055 void MainGame::calculateFPS()
1056 {
1057  static const int NUM_SAMPLES = 1000;
1058  static float frameTimes[NUM_SAMPLES];
1059  static int currentFrame = 0;
1060  static float prevTicks = SDL_GetTicks();
1061 
1062  float currentTicks;
1063  currentTicks = SDL_GetTicks();
1064 
1065  frameTime_ = currentTicks - prevTicks;
1066 
1067  //avoid bug that set frameTime_ to very low negative number on first calculation
1068  if(frameTime_ < 0)
1069  frameTime_ = 0;
1070 
1071  frameTimes[currentFrame % NUM_SAMPLES] = frameTime_;
1072 
1073  // set previous ticks to new ticks now
1074  prevTicks = currentTicks;
1075 
1076  currentFrame++;
1077 
1078  int count;
1079 
1080  if (currentFrame < NUM_SAMPLES)
1081  {
1082  count = currentFrame;
1083  }
1084  else
1085  {
1086  count = NUM_SAMPLES;
1087  }
1088 
1089  float frameTimeAverage = 0;
1090  for (int i = 0; i < count; i++)
1091  {
1092  frameTimeAverage += frameTimes[i];
1093  }
1094  frameTimeAverage /= count;
1095 
1096  if (frameTimeAverage > 0)
1097  {
1098  fps_ = 1000.0 / frameTimeAverage;
1099  }
1100  else
1101  {
1102  fps_ = 60.0f;
1103  }
