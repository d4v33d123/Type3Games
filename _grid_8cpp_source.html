315  int Grid::calcDeathChance(int row, int col, int parentDchance, bool cancerous)
316  {
317  // check distance to closest blood vessel
318  float inRange = false;
319  int distToBv = 999; // high number to make sure first if is true
320  int death_chance; // death chance
321  int bloodvessel_range = T3E::BloodVessel::getRange();
322 
323  for(std::vector<Hex*>::iterator bvs = bloodVessels_.begin(); bvs != bloodVessels_.end(); ++bvs)
324  {
325  BloodVessel* bloodVessel = (BloodVessel*)((*bvs)->getNode());
326  int dist = getDistance((*bvs)->getRow(), (*bvs)->getCol(), row, col);
327  if(dist < distToBv)
328  distToBv = dist;
329  }
330 
331  if( cancerous )
332  {
333  if( distToBv <= bloodvessel_range )
334  {
335  death_chance = cancerDeathChance_;
336  }
337  else
338  {
339  death_chance = clampPercnt(cancerDeathChance_ * (distToBv - bloodvessel_range));
340  }
341  }
342  else
343  {
344  if( distToBv <= 2 )
345  {
346  // The hex is adjacent to the blood vessel
347  death_chance = adjacentBloodvesselDeathChance_;
348  }
349  else if( distToBv <= bloodvessel_range )
350  {
351  // take account of adjacent cells are two away from the blood vessel
352  float lerpAmount = (distToBv - 2) / float(bloodvessel_range - 2);
353 
354  // As the cell gets farther away from the blood vessel
355  // Lerp from the adjacent_death_chance to the far_death_chance
356  death_chance = (adjacentBloodvesselDeathChance_ * (1.0f - lerpAmount)) + (farBloodvesselDeathChance_ * lerpAmount);
357  }
358  else
359  {
360  // If the cell is outside the range of a blood vessel
361  // It's death chance is based on it's parents + some value
362  death_chance = parentDchance + childDeathChanceIncrease_;
363  }
364  }
365 
366  //cap death_chance
367  if(death_chance < minDeathChance_ ) death_chance = minDeathChance_;
368  if(death_chance > maxDeathChance_ ) death_chance = maxDeathChance_;
369 
370  return death_chance;
371  }
