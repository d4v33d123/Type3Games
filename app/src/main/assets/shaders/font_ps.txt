// set default float precision to mediump, highp is not supported on all devices in frag stage
precision mediump float;

// Input parameters
varying vec3 fTexCoord;

uniform sampler2D font_bitmap;

void main()
{
 	vec4 tex = texture2D( font_bitmap, fTexCoord.xy );

	gl_FragColor = vec4( tex.rgb, tex.a * fTexCoord.z);
}
