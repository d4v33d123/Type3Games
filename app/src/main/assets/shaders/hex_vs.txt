//vertex shader
//transform vertex to homogeneous space

//input constants
uniform lowp vec4 inputColour;
uniform mat4 finalM;

//input variables
attribute vec4 aPosition;
//attribute lowp vec4 aColour;

//output variables
varying lowp vec4 vColour;

void main() 
{
	//transform position with world view and projection matrices
	gl_Position = finalM * aPosition;

	//pass on unchanged colour
    vColour = inputColour;
}
